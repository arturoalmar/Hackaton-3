public class Main {
    public static void main(String[] args) {
        // Creo una matriz 5x5
        int[][] matriz = new int[5][5];
                //Ahora la muestro por pantalla
        for(int i = 0; i < 5; i++) {
            for (int j = 0; j < 5; j++) {
                System.out.println(matriz[j][i] + " ");
            }
            System.out.println();
        }

    }
